Lambdas in C++11

// Usage: g++ -std=c++11

// References:
// http://www.drdobbs.com/cpp/lambdas-in-c11/240168241

#include <iostream>
using namespace std;


int sum(int x, int y) { return x + y; }
void sumEx() {
      cout << sum(5, 2) << endl;
      cout << sum(10, 5) << endl;  
}

// Annonymous function 
void sumEx1() {
      auto sum = [](int x, int y) { return x + y; };
      cout << sum(5, 2) << endl;
      cout << sum(10, 5) << endl;
}

// Partial function application.
// Uncurried form. See https://wiki.haskell.org/Currying
// The lambda bound variable x appears in the capture list of the inner lambda.
void sumEx2() {
      auto sum = [](int x) { return [x](int y) { return x + y; }; };
      cout << sum(5)(2) << endl;
      cout << sum(10)(5) << endl;
}

// Just playing.
// Note that sum appears in the capture list of inc.
void sumEx3() {
      auto sum = [](int x) { return [x](int y) { return x + y; }; };
      auto inc = [sum](int x) { return sum(x)(1); };
      cout << sum(5)(inc(1)) << endl;
      cout << sum(10)(5) << endl;
}

int main() {
  sumEx();
  sumEx1();
  sumEx2();
  sumEx3();

}
